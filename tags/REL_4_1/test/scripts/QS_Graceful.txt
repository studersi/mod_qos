INCLUDE scripts/ports

CLIENT
_REQ localhost $QS_PORT_BASE
__GET /cgi/sleep.cgi?s=5 HTTP/1.1
__Host: localhost
__Cookie: MODQOS=$COOKIE_VAL
__Connection: keep-alive
__
_EXPECT . "200 OK"
_WAIT
_CLOSE

CLIENT
_SLEEP 500
_EXEC ./ctl.sh graceful >/dev/null
END
