INCLUDE scripts/ports

CLIENT  30
_LOOP 15
_REQ server1 $QS_PORT_BASE1
__GET /cgi/slow.cgi HTTP/1.1
__Host: server1
__Cookie: MODQOS=$COOKIE_VAL
__Connection: keep-alive
__
_EXPECT . "200 OK"
_EXPECT . "long line"
_WAIT
_CLOSE
_END LOOP
END

CLIENT
_SLEEP 4000
_EXEC ./ctl.sh graceful >/dev/null
END
