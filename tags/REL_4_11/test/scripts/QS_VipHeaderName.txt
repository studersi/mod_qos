INCLUDE scripts/ports

# consume all connections ...
CLIENT 5 1
_SLEEP 1000
_REQ localhost $QS_PORT_BASE
__GET /cgi/sleep.cgi?s=10 HTTP/1.1
__Host: localhost
__
_EXPECT . "200 OK"
_WAIT

END

# test vip access
CLIENT
# get a cookie
_REQ localhost $QS_PORT_BASE
__GET /login/vip.cgi HTTP/1.1
__Host: localhost
__
_MATCH headers "MODQOS=(.*); Path" COOKIE_VAL 
_EXPECT . "200 OK"
_WAIT


_SLEEP 2000
# std user
_REQ localhost $QS_PORT_BASE
__GET /cgi/index.cgi HTTP/1.1
__Host: localhost
__
_EXPECT . "500 Internal Server Error"
_WAIT
_CLOSE

# vip user
_REQ localhost $QS_PORT_BASE
__GET /cgi/index.cgi HTTP/1.1
__Host: localhost
__Cookie: MODQOS=$COOKIE_VAL
__
_EXPECT . "200 OK"
_WAIT

# invalid cred
_REQ localhost $QS_PORT_BASE
__GET /cgi/index.cgi HTTP/1.1
__Host: localhost
__Cookie: MODQOS=eg3LcsEBcTAlxjE12JVz+Q/GqaT/PiZC88AOPK08ckQ
__
_EXPECT . "500 Internal Server Error"
_WAIT
_CLOSE

# timeout
_REQ localhost $QS_PORT_BASE
__GET /cgi/index.cgi HTTP/1.1
__Host: localhost
__Cookie: MODQOS=$COOKIE_VAL
__
_EXPECT . "200 OK"
_WAIT

_SLEEP 4000
_REQ localhost $QS_PORT_BASE
__GET /cgi/index.cgi HTTP/1.1
__Host: localhost
__Cookie: MODQOS=$COOKIE_VAL
__
_EXPECT . "500 Internal Server Error"
_WAIT

END
