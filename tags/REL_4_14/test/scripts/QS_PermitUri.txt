CLIENT
_REQ localhost $QS_PORT_BASE

__Host: localhost
__
_EXPECT . "aaa index.html"
_WAIT

_REQ localhost $QS_PORT_BASE
__GET /aaa/index%20.html HTTP/1.1
__Host: localhost
__
_EXPECT . "403 Forbidden"
_WAIT

END
