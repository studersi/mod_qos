INCLUDE scripts/ports

CLIENT 5 1
_REQ localhost $QS_PORT_BASE
__GET /cgi/sleep.cgi?s=10 HTTP/1.1
__Host: localhost
__
_EXPECT . "200 OK"
_WAIT

END

CLIENT 1 1
_SLEEP 2000
_REQ localhost $QS_PORT_BASE
__GET /cgi/sleep.cgi?s=10 HTTP/1.1
__Host: localhost
__
_EXPECT . "500 Internal Server Error"
_EXPECT . "custom error page"
_WAIT

END
